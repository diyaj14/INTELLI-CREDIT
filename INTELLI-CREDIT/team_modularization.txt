  from modules.document_intelligence.document_pipeline import run_pipeline
  from modules.research_agent.research_pipeline import run_research
  from modules.scoring.scoring_pipeline import run_scoring

  DEMO_MODE = st.sidebar.toggle("DEMO MODE", value=True)

  if DEMO_MODE:
      doc_report   = load_json("demo_data/apex_textiles_document.json")
      res_report   = load_json("demo_data/apex_textiles_research.json")
      score_result = load_json("demo_data/apex_textiles_scoring.json")
  else:
      uploaded     = st.file_uploader(...)
      doc_report   = run_pipeline(uploaded, company_name)
      res_report   = run_research(company_name, promoter_names)
      score_result = run_scoring(doc_report, res_report, officer_notes)
